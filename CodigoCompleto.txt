Sub ComplementarInformacion()
    ' Definir las rutas de los archivos
    Dim rutaArchivo1 As String
    Dim rutaArchivo2 As String
    
    rutaArchivo1 = "Ruta_del_Archivo1.xlsx" ' Actualiza con la ruta correcta del primer archivo
    rutaArchivo2 = "Ruta_del_Archivo2.xlsx" ' Actualiza con la ruta correcta del segundo archivo
    
    ' Abrir los archivos
    Workbooks.Open rutaArchivo1
    Workbooks.Open rutaArchivo2
    
    ' Complementar la información del segundo archivo en el primero
    Dim wb1 As Workbook
    Dim wb2 As Workbook
    Dim ws1 As Worksheet
    Dim ws2 As Worksheet
    Dim lastRow As Long
    Dim i As Long
    
    Set wb1 = ActiveWorkbook ' Archivo1
    Set wb2 = Workbooks(2)   ' Archivo2
    
    Set ws1 = wb1.Sheets(1)  ' Hoja 1 del Archivo1
    Set ws2 = wb2.Sheets(1)  ' Hoja 1 del Archivo2
    
    lastRow = ws1.Cells(ws1.Rows.Count, "A").End(xlUp).Row
    
    For i = 2 To lastRow ' Comienza desde la segunda fila (suponiendo que la primera fila son encabezados)
        Dim buscarValor As Variant
        buscarValor = ws1.Cells(i, "A").Value
        
        Dim complementarValor As Variant
        complementarValor = Application.VLookup(buscarValor, ws2.Range("A:B"), 2, False)
        
        If Not IsError(complementarValor) Then ' Comprueba si se encontró un valor coincidente
            ws1.Cells(i, "B").Value = complementarValor ' Complementa el valor en la columna B del Archivo1
        End If
    Next i
    
    ' Cerrar y guardar los archivos
    wb1.Close SaveChanges:=True
    wb2.Close SaveChanges:=False
    
    MsgBox "La información se ha complementado correctamente."
End Sub

'===================================================================
' Exportar_Descargar_Codigo_Completo.bas
' Exporta todos los módulos/formularios/clases, guarda propiedades,
' concatena el código y crea un .zip descargable con todo.
' Autor: ChatGPT (versión VBA)
' Fecha: 2025-11-06
' Requisito: "Confiar en el acceso al modelo de objetos del proyecto VBA"
'===================================================================

Option Explicit

Sub ExportarYDescargar_CodigoYPropiedades()
    Dim CarpetaDestino As String
    Dim vbComp As Object
    Dim NombreArchivo As String
    Dim RutaArchivo As String
    Dim FSO As Object
    Dim ArchivoDestino As String
    Dim ArchivoOut As Object
    Dim ArchivoIn As Object
    Dim Archivo As String
    Dim Texto As String
    Dim Extensiones As Variant
    Dim Ext As Variant
    Dim propiedadesTxt As String
    Dim shellApp As Object
    Dim ZipPath As String
    
    On Error GoTo ErrHandler
    
    ' ---- 1) Seleccionar carpeta destino ----
    With Application.FileDialog(msoFileDialogFolderPicker)
        .Title = "Seleccione la carpeta donde se guardarán los archivos exportados"
        If .Show <> -1 Then Exit Sub
        CarpetaDestino = .SelectedItems(1)
    End With
    If Right(CarpetaDestino, 1) <> "\" Then CarpetaDestino = CarpetaDestino & "\"
    
    Set FSO = CreateObject("Scripting.FileSystemObject")
    
    ' ---- 2) Exportar todos los componentes VBA ----
    For Each vbComp In ThisWorkbook.VBProject.VBComponents
        Select Case vbComp.Type
            Case 1: NombreArchivo = vbComp.Name & ".bas" ' Módulo estándar
            Case 2: NombreArchivo = vbComp.Name & ".cls" ' Clase
            Case 3: NombreArchivo = vbComp.Name & ".frm" ' Formulario
            Case Else: NombreArchivo = vbComp.Name & ".txt"
        End Select
        
        RutaArchivo = CarpetaDestino & NombreArchivo
        vbComp.Export RutaArchivo
    Next vbComp
    
    ' ---- 3) Guardar propiedades del libro en Properties.txt ----
    propiedadesTxt = ObtenerPropiedadesLibroTexto(ThisWorkbook)
    ArchivoDestino = CarpetaDestino & "Properties.txt"
    Set ArchivoOut = FSO.CreateTextFile(ArchivoDestino, True)
    ArchivoOut.WriteLine propiedadesTxt
    ArchivoOut.Close
    
    ' ---- 4) Crear CodigoCompleto.txt uniendo todo ----
    ArchivoDestino = CarpetaDestino & "CodigoCompleto.txt"
    Set ArchivoOut = FSO.CreateTextFile(ArchivoDestino, True)
    
    Extensiones = Array("bas", "cls", "frm", "txt")
    For Each Ext In Extensiones
        Archivo = Dir(CarpetaDestino & "*." & Ext)
        Do While Archivo <> ""
            ' Evitar re-lectura del CodigoCompleto.txt y Properties.txt en bucle
            If LCase(Archivo) <> "codigocompleto.txt" And LCase(Archivo) <> "properties.txt" Then
                Set ArchivoIn = FSO.OpenTextFile(CarpetaDestino & Archivo, 1)
                Texto = ArchivoIn.ReadAll
                ArchivoOut.WriteLine String(60, "'")
                ArchivoOut.WriteLine "Archivo: " & Archivo
                ArchivoOut.WriteLine String(60, "'")
                ArchivoOut.WriteLine Texto
                ArchivoOut.WriteLine vbCrLf & vbCrLf
                ArchivoIn.Close
            End If
            Archivo = Dir
        Loop
    Next Ext
    ArchivoOut.Close
    
    ' ---- 5) Crear ZIP con todo (CodigoCompleto.zip) ----
    ZipPath = CarpetaDestino & "CodigoCompleto.zip"
    ' Crear zip vacío (cabecera mínima) si no existe
    If Not FSO.FileExists(ZipPath) Then
        CrearZipVacio ZipPath
    End If
    
    ' Usar Shell para copiar los archivos al zip
    Set shellApp = CreateObject("Shell.Application")
    Dim sourceFolder As Object
    Set sourceFolder = shellApp.Namespace(CarpetaDestino)
    Dim zipFolder As Object
    Set zipFolder = shellApp.Namespace(ZipPath)
    
    If Not zipFolder Is Nothing Then
        ' Copiar todos los archivos exportados + Properties.txt + CodigoCompleto.txt
        Dim itemsToCopy As Object
        Set itemsToCopy = sourceFolder.Items
        zipFolder.CopyHere itemsToCopy, 4 + 16 ' No mostrar diálogo, NO sobrescribir sin preguntar
        ' Esperar un poco para que termine la copia (simple espera)
        Dim t As Double: t = Timer
        Do While Timer < t + 1.5 ' espera 1.5 segundos (puede ajustarse si hay muchos archivos)
            DoEvents
        Loop
    Else
        MsgBox "No se pudo acceder al zip con Shell. El .zip se creó pero no se llenó.", vbExclamation
    End If
    
    ' ---- 6) Mensaje final con rutas ----
    MsgBox "? Proceso completado." & vbCrLf & vbCrLf & _
           "Archivos exportados en: " & CarpetaDestino & vbCrLf & _
           "- Properties.txt (propiedades del libro)" & vbCrLf & _
           "- CodigoCompleto.txt (todo el código combinado)" & vbCrLf & _
           "- Archivo zip: " & ZipPath, vbInformation, "Finalizado"
    
    Exit Sub

ErrHandler:
    MsgBox "?? Error: " & Err.Number & " - " & Err.Description, vbExclamation, "Error"
End Sub

Attribute VB_Name = "Module2"
Sub ExportarTodoElCodigo()
    Dim CarpetaDestino As String
    Dim ArchivoDestino As String
    Dim vbComp As Object
    Dim FSO As Object
    Dim ArchivoOut As Object
    Dim Texto As String
    Dim RutaTemp As String
    Dim ArchivoTemp As String
    
    On Error GoTo ErrorHandler
    
    ' --- Seleccionar carpeta destino ---
    With Application.FileDialog(msoFileDialogFolderPicker)
        .Title = "Selecciona una carpeta donde guardar el código exportado"
        If .Show <> -1 Then Exit Sub
        CarpetaDestino = .SelectedItems(1)
    End With
    
    If Right(CarpetaDestino, 1) <> "\" Then CarpetaDestino = CarpetaDestino & "\"
    
    ' --- Crear archivo final ---
    ArchivoDestino = CarpetaDestino & "CodigoCompleto.txt"
    Set FSO = CreateObject("Scripting.FileSystemObject")
    Set ArchivoOut = FSO.CreateTextFile(ArchivoDestino, True)
    
    ' --- Recorrer todos los módulos ---
    For Each vbComp In ThisWorkbook.VBProject.VBComponents
        RutaTemp = CarpetaDestino & vbComp.Name & ".txt"
        
        ' Exportar módulo temporal
        vbComp.Export RutaTemp
        
        ' Leer el contenido exportado
        With FSO.OpenTextFile(RutaTemp, 1)
            Texto = .ReadAll
            .Close
        End With
        
        ' Escribir en el archivo combinado
        ArchivoOut.WriteLine String(80, "=")
        ArchivoOut.WriteLine ">>> Módulo: " & vbComp.Name
        ArchivoOut.WriteLine String(80, "=")
        ArchivoOut.WriteLine Texto
        ArchivoOut.WriteLine vbCrLf
    Next vbComp
    
    ' --- Cerrar archivo y limpiar ---
    ArchivoOut.Close
    
    MsgBox "? Código exportado correctamente." & vbCrLf & _
           "Archivo generado en:" & vbCrLf & ArchivoDestino, vbInformation

    Exit Sub

ErrorHandler:
    MsgBox "?? Error al exportar: " & Err.Description, vbCritical
End Sub
